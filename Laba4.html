Варіант 10: Перегляд набору зображень зі зміною підписів до зображень за допомогою кнопок «Назад» та «Далі».

    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <title>Перегляд зображень</title>
    <style>
        body {
            font-family: Arial, sans-serif;
            text-align: center;
        }

        #imageContainer {
            width: 80%;
            margin: 20px auto;
        }

        #image {
            max-width: 100%;
            height: auto;
        }

        #caption {
            margin-top: 10px;

        }

<body>

    <h1>Перегляд зображень</h1>

    <div id="imageContainer">
        <img id="image" src="img1.jpg" alt="Зображення 1">
        <div id="caption">Зображення 1: Опис цього зображення.</div>
    </div>

    <div class="buttons">
        <button onclick="changeImage(-1)">Назад</button>
        <button onclick="changeImage(1)">Далі</button>
    </div>

    <script>
        const images = [
            { src: "img1.jpg", caption: "Зображення 1: Гарна картинка." },
            { src: "img2.jpg", caption: "Зображення 2: Ще одна гарна картинка." },
            { src: "img3.jpg", caption: "Зображення 3: Третя гарна картинка." },          
        ];

        let currentIndex = 0;

        function changeImage(direction) {
            currentIndex += direction;
            if (currentIndex < 0) {
                currentIndex = images.length - 1;
            } else if (currentIndex >= images.length) {
                currentIndex = 0;
            }

            const image = images[currentIndex];
            document.getElementById("image").src = image.src;
            document.getElementById("caption").textContent = image.caption;
        }
    </script>
